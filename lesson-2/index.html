var a = 1, b = 1, c, d;

c = ++a; alert(c); // 2 - эта форма инкремента называется префиксная, т.е. она прибавляет единицу к переменной 'а'.

d = b++; alert(d); // 1 - это форма тоже прибавляет единицу к переменной, но прежде, она запишет старое значение в переменную 'd'.

c = (2+ ++a); alert(c); // 5 - в этом примере у переменной 'a' уже значение 2, и к этой 2ке прибавляем ещё 1 получается 3, а 2 + 3 получается 5.

d = (2+ b++); alert(d); // 4 - а сдесь у переменной 'b' вернулось новое значение из 2го примера, т.е. теперь у неё будет не 1, а 2, и 2 + 2 = 4.

alert(a); // 3 - переменная 'а' вызывалась 2 раза, и все 2 раза к ней прибавляли единицу.

alert(b); // 3 - переменную 'b' тоже вызывали 2 раза и 2 раза к ней прибавляли единецу.
